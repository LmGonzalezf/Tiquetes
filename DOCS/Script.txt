CREATE TABLE "usuarios" (
  "id" SERIAL PRIMARY KEY ,
  "nombre" varchar(30) NOT NULL,
  "taquilla" int NOT NULL,
  "contrasena" varchar(20) NOT NULL
);

CREATE TABLE "taquillas" (
  "id" SERIAL PRIMARY KEY,
  "nombre" varchar(20) NOT NULL
);

CREATE TABLE "rutas" (
  "id_origen" int ,
  "id_destino" int ,
  "nombre" varchar(30),
  "precio" int,
	PRIMARY KEY (id_origen, id_destino)	
);

CREATE TABLE "tiquetes" (
  "id" SERIAL PRIMARY KEY,
  "cedula" int NOT NULL,
  "id_linea" int NOT NULL,
  "fecha_exp" date NOT NULL,
  "hora_exp" date NOT NULL,
  "fecha" date NOT NULL,
  "hora" time NOT NULL,
  "cantidad_sillas" int NOT NULL,
  "precio" int NOT NULL,
  "origen" int,
  "destino" int,
  "vendedor" int
);

CREATE TABLE "destinos" (
  "id" int PRIMARY KEY,
  "nombre" varchar(30)
);

CREATE TABLE "clientes" (
  "cedula" int PRIMARY KEY,
  "nombre" varchar(20) NOT NULL,
  "celular" int
);

CREATE TABLE "cierres" (
  "id" SERIAL PRIMARY KEY,
  "totalRionegro" int,
  "totalgomezVilla" int,
  "usuario" int,
  "total" int,
  "fecha" date,
  "hora" time
);

CREATE TABLE "remesas" (
  "id" SERIAL PRIMARY KEY,
  "id_linea" int NOT NULL,
  "origen" int NOT NULL,
  "destino" int NOT NULL,
  "contiene" varchar(200) NOT NULL,
  "cedula_remitente" int NOT NULL,
  "cedula_destinatario" int NOT NULL,
  "fecha" date NOT NULL,
  "vendedor" int
);

CREATE TABLE "carros" (
  "numero" int PRIMARY KEY,
  "empresa" int NOT NULL,
  "placa" varchar(10) NOT NULL,
  "tipo" int NOT NULL,
  "num_puestos" int NOT NULL,
  "conductor" int NOT NULL
);

CREATE TABLE "tipos" (
  "tipo" int PRIMARY KEY,
  "nombre" varchar(20) NOT NULL
);

CREATE TABLE "lineas" (
  "id" SERIAL PRIMARY KEY,
  "num_carro" int NOT NULL,
  "hora" time NOT NULL,
  "fecha" date NOT NULL,
  "destino" int NOT NULL,
  "origen" int NOT NULL,
  "estado" int NOT NULL,
  "total" int,
  "totalTiquetes" int
);

CREATE TABLE "estados" (
  "id" int PRIMARY KEY,
  "nombre" varchar(20) NOT NULL
);

CREATE TABLE "conductores" (
  "cedula" int PRIMARY KEY,
  "nombre" varchar(30) NOT NULL
);

CREATE TABLE "empresa" (
  "id" int PRIMARY KEY,
  "nombre" varchar(30) NOT NULL
);

CREATE TABLE "linea_tiquetes" (
  "id_linea" int,
  "id_tiquete" int,
  "num_puesto" int,
	UNIQUE (id_linea, num_puesto)
);
